Selection Queries

SELECT *
FROM BEER
WHERE tap = true;

/*Displays all beers that have a A.B.V. (Alcohol by Volume) greater than 5% */
SELECT *
FROM BEER
WHERE volume > 5 AND tap = true;

SELECT name
FROM GAMES
WHERE (description LIKE '%hockey%') OR (description LIKE '%basketball%') OR (description LIKE '%football%')
