SQL Schema

Here are the SQL commands to create the tables in our database.

DROP TABLE IF EXISTS EVENTS;
DROP TABLE IF EXISTS BEER;
DROP TABLE IF EXISTS GAMES;

CREATE TABLE EVENTS(
    name text, 
    description text, 
    startDate date, 
    private boolean
);

CREATE TABLE BEER(
    name text, 
    type text, 
    tap boolean, 
    volume decimal
);

CREATE TABLE GAMES(
    name text, 
    description text
);
